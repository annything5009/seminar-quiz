ผู้เล่นเข้าที่: http://localhost:3000/player.html"); console.log("======================================"); // ---------- สถานะเกม ---------- let phase = "lobby"; // lobby | question | reveal | over let currentIndex = -1; let questionStartAt = 0; let questionTimer = null; const players = new Map(); // socketId -> { name, score, answered, answerIndex, timeMs } const hostSockets = new Set(); function playerList() { return [...players.values()]; } function top3() { return playerList() .sort((a, b) => b.score - a.score) .slice(0, 3) .map((p) => ({ name: p.name, score: p.score })); } function broadcastLobbyCount() { hostSockets.forEach((id) => io.to(id).emit("host:lobby", { count: players.size })); } function sendQuestion(index) { currentIndex = index; phase = "question"; questionStartAt = Date.now(); players.forEach((p) => { p.answered = false; p.answerIndex = null; p.timeMs = null; }); const q = questions[index]; io.to("players").emit("question", { index, total: questions.length, question: q.question, choices: q.choices, timeLimit: q.timeLimit, }); hostSockets.forEach((id) => io.to(id).emit("host:question", { index, total: questions.length, question: q.question, choices: q.choices, totalPlayers: players.size, }) ); clearTimeout(questionTimer); questionTimer = setTimeout(() => revealQuestion(), q.timeLimit * 1000 + 300); } function revealQuestion() { if (phase !== "question") return; clearTimeout(questionTimer); phase = "reveal"; const q = questions[currentIndex]; const distribution = [0, 0, 0, 0]; players.forEach((p, socketId) => { if (p.answered && p.answerIndex !== null) { distribution[p.answerIndex] = (distribution[p.answerIndex] || 0) + 1; } const correct = p.answered && p.answerIndex === q.correctIndex; let scoreGained = 0; if (correct) { const remaining = Math.max(0, q.timeLimit * 1000 - p.timeMs); const fraction = remaining / (q.timeLimit * 1000); scoreGained = Math.round(500 + 500 * fraction); p.score += scoreGained; } io.to(socketId).emit("reveal", { yourAnswer: p.answered ? p.answerIndex : null, correct, scoreGained, totalScore: p.score, }); }); const isLast = currentIndex + 1 >= questions.length; hostSockets.forEach((id) => io.to(id).emit("host:reveal", { isLast, distribution, correctIndex: q.correctIndex, top3: top3(), }) ); } function nextQuestion() { const nextIndex = currentIndex + 1; if (nextIndex >= questions.length) { phase = "over"; io.to("players").emit("gameOver"); hostSockets.forEach((id) => io.to(id).emit("host:gameOver", { top3: top3() })); return; } sendQuestion(nextIndex); } function resetGame() { clearTimeout(questionTimer); phase = "lobby"; currentIndex = -1; players.forEach((p) => (p.score = 0)); io.to("players").emit("resetToLobby"); broadcastLobbyCount(); } // ---------- socket.io ---------- io.on("connection", (socket) => { // ---- ผู้เล่น ---- socket.on("join", ({ name }) => { players.set(socket.id, { name: (name || "ผู้เล่น").slice(0, 24), score: 0, Math.max(1, q.timeLimit - Math.floor(elapsed / 1000)), }); } }); socket.on("answer", ({ choiceIndex }) => { const p = players.get(socket.id); if (!p || phase !== "question" || p.answered) return; p.answered = true; p.answerIndex = choiceIndex; p.timeMs = Date.now() - questionStartAt; const answeredCount = playerList().filter((pl) => pl.answered).length; hostSockets.forEach((id) => io.to(id).emit("host:answerProgress", { answeredCount, totalPlayers: players.size }) ); }); // ---- โฮสต์ ---- socket.on("host:auth", ({ key }, cb) => { if (key === HOST_KEY) { hostSockets.add(socket.id); cb({ ok: true }); socket.emit("host:lobby", { count: players.size }); } else { cb({ ok: false }); } }); socket.on("host:start", () => { if (!hostSockets.has(socket.id)) return; sendQuestion(0); }); socket.on("host:next", () => { if (!hostSockets.has(socket.id)) return; nextQuestion(); }); socket.on("host:reset", () => { if (!hostSockets.has(socket.id)) return; resetGame(); }); socket.on("disconnect", () => { hostSockets.delete(socket.id); if (players.delete(socket.id)) broadcastLobbyCount(); }); }); const PORT = process.env.PORT || 3000; server.listen(PORT,
