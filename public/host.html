---------- สถานะเกม ---------- let phase = "lobby"; // lobby | question | reveal | over let currentIndex = -1; let questionStartAt = 0; let questionTimer = null; const players = new Map(); // socketId -> { name, score, answered, answerIndex, timeMs } const hostSockets = new Set(); function playerList() { return [...players.values()]; } function top3() { return playerList() .sort((a, b) => b.score - => revealQuestion(), q.timeLimit * 1000 + 300); } function revealQuestion() { if (phase !== "question") return; clearTimeout(questionTimer); phase = "reveal"; const q = questions[currentIndex]; const distribution = [0, 0, 0, 0]; players.forEach((p, socketId) => { if (p.answered && p.answerIndex !== null) { distribution[p.answerIndex] = (distribution[p.answerIndex] || 0) + 1; } const correct = }); const isLast = currentIndex + 1 >= questions.length; hostSockets.forEach((id) => io.to(id).emit("host:reveal", { isLast, distribution, correctIndex: q.correctIndex, top3: top3(), }) ); } function nextQuestion() { const nextIndex = currentIndex + 1; if (nextIndex >= questions.length) { phase = "over"; io.to("players").emit("gameOver"); hostSockets.forEach((id) => io.to(id).emit("host:gameOver", { top3: top3() })); return; } sendQuestion(nextIndex); } function resetGame() { clearTimeout(questionTimer); phase = "lobby"; currentIndex = -1; players.forEach((p) => (p.score = 0)); } }); socket.on("answer", ({ choiceIndex }) => { const p = players.get(socket.id); if (!p || phase !== "question" || p.answered) return; p.answered = true; p.answerIndex = choiceIndex; p.timeMs =
